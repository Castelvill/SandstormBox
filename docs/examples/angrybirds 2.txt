let game = obj.game.var;
let money = me.var.money;

each_iteration{
	let walls = all(group=="walls");
	let me_walls = first(walls.collision_with == me)
	if(me_walls.size() > 0){
		game.state = 0;
	}
	let money_coll = all(group=="money" && collision_with == me);
	money += money_coll.val.value;
	money_coll.kill();
	
	walls = all(walls.collision_with == objects.border);
	walls.x += game.jumpWalls;
	let_vec SIZE = rand(game.wallMin, game.wallMax);
	walls.events.setHeight(SIZE);

    	let lvlCup = me.var.levelCup;
    	if(money >= lvlCup){
        	lvlCup *= 1.5;
        	let t1 = clone(objects.Toast);
        	t1.text.label = "Level up!";
        	t1.events.run_toast();
    	}
}
